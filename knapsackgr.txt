//knapsack



#include <stdio.h>



void main() {
int cur_w;
float tot_v;
int i, maxi;
int used[10];



int n,W,w[10],p[10];
printf("Enter number of objects: ");
scanf("%d",&n);
printf("Enter profit of objects: ");
for(int j=0;j<n;j++)
scanf("%d",&p[j]);
printf("Enter weight of the objects: ");
for(int j=0;j<n;j++)
scanf("%d",&w[j]);
printf("Enter the maximum weight the bag can take: ");
scanf("%d",&W);
for (i = 0; i < n; ++i)
used[i] = 0;



cur_w = W;
while (cur_w > 0) {
maxi = -1;
for (i = 0; i < n; ++i)
if ((used[i] == 0) && ((maxi == -1) || ((float)p[i]/w[i] > (float)p[maxi]/w[maxi])))
maxi = i;



used[maxi] = 1; /* mark the maxi-th object as used */
cur_w -= w[maxi]; /* with the object in the bag, I can carry less */
tot_v += p[maxi];
if (cur_w >= 0)
printf("Added object %d (%d profit, %d weight) completely in the bag. Space left: %d.\n", maxi + 1, p[maxi], w[maxi], cur_w);
else {
printf("Added %d%% (%d profit, %d weight) of object %d in the bag.\n", (int)((1 + (float)cur_w/w[maxi]) * 100), p[maxi], w[maxi], maxi + 1);
tot_v -= p[maxi];
tot_v += (1 + (float)cur_w/w[maxi]) * p[maxi];
}
}



printf("Filled the bag with objects worth %.2f profit.\n", tot_v);
}
